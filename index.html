package main

      import (
          "fmt"
          "net/http"
      
          "github.com/gin-gonic/gin"
      )
      
      func main() {
          router := gin.Default()
      
          // Define a POST route for the webhook endpoint
          router.POST("/webhook", func(c *gin.Context) {
              // Parse the request body into an event object
              var event struct {
                  Type string `json:"type"`
              }
              if err := c.ShouldBindJSON(&event); err != nil {
                  c.AbortWithStatusJSON(http.StatusBadRequest, gin.H{"error": "Invalid JSON payload"})
                  return
              }
      
              // Handle the event based on its type
              switch event.Type {
              case "test":
                  fmt.Println("Received test event")
              default:
                  fmt.Println("Received unknown event type:", event.Type)
              }
      
              // Send a 2xx response status code to indicate success
              c.JSON(http.StatusOK, gin.H{})
          })
      
          // Start the server
          if err := router.Run(":8080"); err != nil {
              panic(err)
          }
      }
